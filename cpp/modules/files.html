Obsługa Plików 📁

Zapis i odczyt danych z plików tekstowych przy użyciu fstream.

1. Biblioteka fstream

Do pracy z plikami potrzebujemy nagłówka <fstream>. Mamy 3 główne klasy:

  • ofstream – zapis do pliku (output)
  • ifstream – odczyt z pliku (input)
  • fstream – zarówno zapis jak i odczyt

2. Zapis do Pliku

#include <fstream>
#include <iostream>
using namespace std;

int main() {
    # Tworzenie pliku do zapisu:
    ofstream plik("dane.txt");
    
    if (plik.is_open()) {
        plik << "Linia 1" << endl;
        plik << "Linia 2" << endl;
        plik << 123 << endl;
        plik.close();
        cout << "Zapisano!" << endl;
    } else {
        cout << "Błąd otwarcia pliku!" << endl;
    }
    
    return 0;
}

3. Odczyt z Pliku

#include <fstream>
#include <iostream>
#include <string>
using namespace std;

int main() {
    ifstream plik("dane.txt");
    string linia;
    
    if (plik.is_open()) {
        # Czytanie linia po linii:
        while (getline(plik, linia)) {
            cout << linia << endl;
        }
        plik.close();
    }
    
    return 0;
}

4. Dopisywanie do Pliku

Domyślnie ofstream nadpisuje plik. Aby dopisywać, użyj flagi ios::app.

# Dopisywanie (app = append):
ofstream plik("log.txt", ios::app);

plik << "Nowa wpisana linia" << endl;
plik.close();

5. 🛠️ Projekt: Dziennik (Logger)

#include <fstream>
#include <iostream>
#include <ctime>
using namespace std;

void loguj(string wiadomosc) {
    ofstream plik("log.txt", ios::app);
    if (plik.is_open()) {
        time_t teraz = time(0);
        plik << ctime(&teraz) << ": " << wiadomosc << endl;
        plik.close();
    }
}

int main() {
    loguj("Program uruchomiony");
    
    string input;
    cout << "Wpisz wiadomość do logu: ";
    getline(cin, input);
    
    loguj(input);
    cout << "Zapisano do log.txt" << endl;
    
    return 0;
}
