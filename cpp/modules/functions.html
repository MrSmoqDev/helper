Funkcje ⚡

Deklaracja, definicja, parametry, zwracanie wartości, przeciążanie.

1. Podstawy Funkcji

# Deklaracja (prototyp):
int dodaj(int a, int b);

int main() {
    int wynik = dodaj(5, 3);
    cout << "Wynik: " << wynik << endl;
    return 0;
}

# Definicja:
int dodaj(int a, int b) {
    return a + b;
}

# Funkcja bez wartości zwracanej (void):
void powitanie() {
    cout << "Witaj!" << endl;
}

powitanie();  # Witaj!

2. Parametry i Zwracanie Wartości

# Wiele parametrów:
double srednia(double a, double b, double c) {
    return (a + b + c) / 3.0;
}

# Parametry domyślne:
void log(string msg, string poziom = "INFO") {
    cout << "[" << poziom << "] " << msg << endl;
}

log("Program startuje");           # [INFO] Program startuje
log("Błąd krytyczny", "ERROR");  # [ERROR] Błąd krytyczny

# Przekazywanie przez wartość (kopia):
void zmien_wartosc(int x) {
    x = 100;  # Zmienia tylko kopię!
}

int a = 5;
zmien_wartosc(a);
cout << a;  # 5 (niezmienione)

3. Przekazywanie przez Referencję

# Referencja (&) - zmiana oryginału:
void zamien(int &a, int &b) {
    int temp = a;
    a = b;
    b = temp;
}

int x = 5, y = 10;
zamien(x, y);
cout << x << " " << y;  # 10 5

# const reference - bez zmian, ale wydajnie:
void wyswietl(const string &tekst) {
    cout << tekst << endl;
    # tekst = "zmiana";  # BŁĄD - const!
}

4. Przeciążanie Funkcji

# Ta sama nazwa, różne parametry:
int dodaj(int a, int b) {
    return a + b;
}

double dodaj(double a, double b) {
    return a + b;
}

int dodaj(int a, int b, int c) {
    return a + b + c;
}

cout << dodaj(2, 3) << endl;        # 5 (int)
cout << dodaj(2.5, 3.1) << endl;  # 5.6 (double)
cout << dodaj(1, 2, 3) << endl;    # 6 (3 parametry)

5. Rekurencja

# Silnia (n!):
int silnia(int n) {
    if (n <= 1) return 1;
    return n * silnia(n - 1);
}

cout << silnia(5);  # 120 (5*4*3*2*1)

# Ciąg Fibonacciego:
int fib(int n) {
    if (n <= 1) return n;
    return fib(n - 1) + fib(n - 2);
}

cout << fib(10);  # 55

6. 🛠️ Projekt: Kalkulator Naukowy

#include <iostream>
#include <cmath>
using namespace std;

double dodaj(double a, double b) { return a + b; }
double odejmij(double a, double b) { return a - b; }
double mnoz(double a, double b) { return a * b; }
double dziel(double a, double b) {
    if (b == 0) {
        cout << "Błąd: Dzielenie przez zero!" << endl;
        return 0;
    }
    return a / b;
}
double potega(double a, double b) { return pow(a, b); }
double pierwiastek(double a) { return sqrt(a); }

int main() {
    double a, b;
    int wybor;
    
    cout << "=== KALKULATOR NAUKOWY ===" << endl;
    cout << "1. Dodawanie\n2. Odejmowanie\n3. Mnożenie\n4. Dzielenie\n5. Potęga\n6. Pierwiastek" << endl;
    cout << "Wybierz: ";
    cin >> wybor;
    
    cout << "Podaj pierwszą liczbę: ";
    cin >> a;
    
    if (wybor != 6) {
        cout << "Podaj drugą liczbę: ";
        cin >> b;
    }
    
    double wynik;
    switch (wybor) {
        case 1: wynik = dodaj(a, b); break;
        case 2: wynik = odejmij(a, b); break;
        case 3: wynik = mnoz(a, b); break;
        case 4: wynik = dziel(a, b); break;
        case 5: wynik = potega(a, b); break;
        case 6: wynik = pierwiastek(a); break;
        default: cout << "Nieprawidłowy wybór!"; return 1;
    }
    
    cout << "Wynik: " << wynik << endl;
    return 0;
}
