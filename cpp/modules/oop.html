Programowanie Obiektowe 🏗️

Klasy, obiekty, enkapsulacja, konstruktory i dziedziczenie.

1. Klasy i Obiekty

Klasa to szablon, a obiekt to konkretna instancja tej klasy.

class Samochod {
public:
    string marka;
    string model;
    int rok;
    
    void jedz() {
        cout << " " << marka << " " << model << " jedzie!" << endl;
    }
};

int main() {
    # Tworzenie obiektów:
    Samochod auto1;
    auto1.marka = "Toyota";
    auto1.model = "Corolla";
    auto1.rok = 2020;
    
    auto1.jedz();  # Toyota Corolla jedzie!
    
    return 0;
}

2. Konstruktory

Specjalna metoda wywoływana automatycznie przy tworzeniu obiektu.

class Osoba {
public:
    string imie;
    int wiek;
    
    # Konstruktor:
    Osoba(string i, int w) {
        imie = i;
        wiek = w;
    }
    
    void przedstaw() {
        cout << "Cześć, mam " << wiek << " lat i mam na imię " << imie << endl;
    }
};

int main() {
    # Wywołanie konstruktora:
    Osoba jan("Jan", 25);
    jan.przedstaw();  # Cześć, mam 25 lat...
    
    return 0;
}

3. Enkapsulacja (private/public)

Ukrywaj dane wewnętrzne (private), udostępniaj metody (public).

class Konto {
private:
    double saldo;  # Niedostępne z zewnątrz
    
public:
    Konto() { saldo = 0; }
    
    void wplac(double kwota) {
        if (kwota > 0) saldo += kwota;
    }
    
    double getSaldo() {
        return saldo;
    }
};

int main() {
    Konto moje;
    moje.wplac(100);
    # moje.saldo = 1000;  # BŁĄD! private
    cout << "Saldo: " << moje.getSaldo() << endl;
    return 0;
}

4. Dziedziczenie

class Zwierzak {
public:
    void oddychaj() { cout << "Oddycham" << endl; }
};

# Klasa pies dziedziczy po Zwierzak:
class Pies : public Zwierzak {
public:
    void szczekaj() { cout << "Hau hau!" << endl; }
};

int main() {
    Pies burek;
    burek.szczekaj();   # Hau hau!
    burek.oddychaj();  # Oddycham (odziedziczone)
    return 0;
}

5. 🛠️ Projekt: System Bohaterów

class Bohater {
protected:
    string imie;
    int hp;
    int atak;
    
public:
    Bohater(string i, int h, int a) : imie(i), hp(h), atak(a) {}
    
    void atakuj(Bohater &cel) {
        cel.hp -= atak;
        cout << imie << " atakuje " << cel.imie << " za " << atak << " obrażeń!" << endl;
    }
    
    bool zyje() { return hp > 0; }
    void status() { cout << imie << " HP: " << hp << endl; }
};

int main() {
    Bohater gracz("Rycerz", 100, 20);
    Bohater wrog("Smok", 150, 30);
    
    while (gracz.zyje() && wrog.zyje()) {
        gracz.atakuj(wrog);
        if (wrog.zyje()) wrog.atakuj(gracz);
    }
    
    if (gracz.zyje()) cout << "Wygrał Gracz!" << endl;
    else cout << "Wygrał Wróg!" << endl;
    
    return 0;
}
