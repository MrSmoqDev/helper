Gra: Kółko i Krzyżyk 🎮

Kompletny projekt konsolowy wykorzystujący tablice, pętle i funkcje.

1. Opis Projektu

Stworzymy klasyczną grę dla dwóch graczy na planszy 3x3. Projekt wykorzystuje:

  • ✅ Tablicę 2D do przechowywania planszy
  • ✅ Pętle do rysowania i sprawdzania
  • ✅ Funkcje do modularności kodu
  • ✅ Warunki zwycięstwa i remisu

2. Pełny Kod Gry

#include <iostream>
using namespace std;

char plansza[3][3] = {
    {'1', '2', '3'},
    {'4', '5', '6'},
    {'7', '8', '9'}
};

char gracz = 'X';

void rysujPlansze() {
    cout << "\n";
    for (int i = 0; i < 3; i++) {
        cout << " " << plansza[i][0] << " | " << plansza[i][1] << " | " << plansza[i][2] << endl;
        if (i < 2) cout << "---|---|---" << endl;
    }
    cout << "\n";
}

void zmianaGracza() {
    gracz = (gracz == 'X') ? 'O' : 'X';
}

bool sprawdzWygrana() {
    # Sprawdzenie wierszy i kolumn
    for (int i = 0; i < 3; i++) {
        if (plansza[i][0] == plansza[i][1] && plansza[i][0] == plansza[i][2]) return true;
        if (plansza[0][i] == plansza[1][i] && plansza[0][i] == plansza[2][i]) return true;
    }
    # Przekątne
    if (plansza[0][0] == plansza[1][1] && plansza[1][1] == plansza[2][2]) return true;
    if (plansza[0][2] == plansza[1][1] && plansza[1][1] == plansza[2][0]) return true;
    
    return false;
}

bool sprawdzRemis() {
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            if (plansza[i][j] != 'X' && plansza[i][j] != 'O') return false;
        }
    }
    return true;
}

int main() {
    int ruch;
    cout << "=== KÓŁKO I KRZYŻYK ===" << endl;
    
    while (true) {
        rysujPlansze();
        cout << "Gracz " << gracz << ", podaj numer pola (1-9): ";
        cin >> ruch;
        
        # Konwersja 1-9 na indeksy tablicy
        int rzad = (ruch - 1) / 3;
        int kol = (ruch - 1) % 3;
        
        if (ruch < 1 || ruch > 9 || plansza[rzad][kol] == 'X' || plansza[rzad][kol] == 'O') {
            cout << "Nieprawidłowy ruch! Spróbuj ponownie." << endl;
            continue;
        }
        
        plansza[rzad][kol] = gracz;
        
        if (sprawdzWygrana()) {
            rysujPlansze();
            cout << "🎉 Gracz " << gracz << " wygrywa!" << endl;
            break;
        }
        
        if (sprawdzRemis()) {
            rysujPlansze();
            cout << "🤝 Remis!" << endl;
            break;
        }
        
        zmianaGracza();
    }
    
    return 0;
}

3. Jak Uruchomić?

  1. Skopiuj kod do pliku gra.cpp.
  2. Otwórz terminal w folderze z plikiem.
  3. Skompiluj: g++ -o gra gra.cpp
  4. Uruchom: ./gra (Linux/Mac) lub gra.exe (Windows).
  5. Graj wpisując numery pól od 1 do 9.
