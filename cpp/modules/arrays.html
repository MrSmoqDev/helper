Tablice i Wektory 📊

Przechowywanie wielu danych: tablice statyczne, dynamiczne wektory i stringi.

1. Tabice Statyczne (Arrays)

Tablica to zbiór elementów tego samego typu pod jednym indeksem.

# Deklaracja tablicy 5 liczb:
int liczby[5] = {10, 20, 30, 40, 50};

# Dostęp (indeks od 0!):
cout << liczby[0] << endl;  # 10
cout << liczby[2] << endl;  # 30

# Zmiana wartości:
liczby[1] = 99;

# Iteracja pętlą:
for (int i = 0; i < 5; i++) {
    cout << liczby[i] << " ";
}

# Pętla zakresowa (C++11):
for (int x : liczby) {
    cout << x << " ";
}

2. Wektory (std::vector)

Wektory to dynamiczne tablice – mogą zmieniać rozmiar w trakcie działania programu.

#include <vector>
using namespace std;

# Tworzenie wektora:
vector<int> liczby;

# Dodawanie elementów:
liczby.push_back(10);
liczby.push_back(20);
liczby.push_back(30);

# Rozmiar:
cout << "Rozmiar: " << liczby.size() << endl;  # 3

# Dostęp:
cout << liczby[0] << endl;  # 10

# Usuwanie ostatniego:
liczby.pop_back();

# Iteracja:
for (int x : liczby) {
    cout << x << " ";
}

3. Stringi (std::string)

W C++ używamy std::string zamiast tablic znaków char[].

#include <string>

string imie = "Jan";
string nazwisko = "Kowalski";

# Łączenie stringów:
string pelne = imie + " " + nazwisko;
cout << pelne << endl;  # Jan Kowalski

# Długość:
cout << imie.length() << endl;  # 3

# Dostęp do znaku:
cout << imie[0] << endl;  # J

# Porównywanie:
if (imie == "Jan") {
    cout << "To Jan!" << endl;
}

4. Tablice Wielowymiarowe

# Tablica 3x3 (np. plansza):
int plansza[3][3] = {
    {0, 0, 0},
    {0, 0, 0},
    {0, 0, 0}
};

# Dostęp:
plansza[1][2] = 5;  # Wiersz 1, Kolumna 2

# Iteracja:
for (int i = 0; i < 3; i++) {
    for (int j = 0; j < 3; j++) {
        cout << plansza[i][j] << " ";
    }
    cout << endl;
}

5. 🛠️ Projekt: Lista Zakupów

#include <iostream>
#include <vector>
#include <string>
using namespace std;

int main() {
    vector<string> lista;
    string item;
    
    cout << "=== LISTA ZAKUPÓW ===" << endl;
    cout << "Wpisuj produkty (wpisz 'koniec' aby zakończyć):" << endl;
    
    while (true) {
        cin >> item;
        if (item == "koniec") break;
        lista.push_back(item);
    }
    
    cout << "\nTwoja lista:" << endl;
    for (int i = 0; i < lista.size(); i++) {
        cout << (i + 1) << ". " << lista[i] << endl;
    }
    
    return 0;
}
