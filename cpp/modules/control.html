Struktury Sterujące 🔀

if/else, switch, pętle for, while, do-while.

1. Instrukcja if/else

int wiek = 18;

if (wiek >= 18) {
    cout << "Pełnoletni" << endl;
} else {
    cout << "Niepełnoletni" << endl;
}

# Wiele warunków:
int wynik = 75;

if (wynik >= 90) {
    cout << "Bardzo dobry" << endl;
} else if (wynik >= 75) {
    cout << "Dobry" << endl;
} else if (wynik >= 50) {
    cout << "Dostateczny" << endl;
} else {
    cout << "Niedostateczny" << endl;
}

2. Instrukcja switch

int dzien = 3;

switch (dzien) {
    case 1:
        cout << "Poniedziałek" << endl;
        break;
    case 2:
        cout << "Wtorek" << endl;
        break;
    case 3:
        cout << "Środa" << endl;
        break;
    default:
        cout << "Inny dzień" << endl;
}

# Bez break - wykonuje się kolejne case'y:
char ocena = 'B';
switch (ocena) {
    case 'A': cout << "Celujący" << endl;
    case 'B': cout << "Bardzo dobry" << endl; # Wykona się!
    default: cout << "Koniec" << endl;
}

3. Pętla for

# Podstawowa pętla for:
for (int i = 0; i < 5; i++) {
    cout << i << " ";  # 0 1 2 3 4
}

# Odliczanie:
for (int i = 10; i > 0; i--) {
    cout << i << " ";  # 10 9 8 ... 1
}

# Krok co 2:
for (int i = 0; i <= 10; i += 2) {
    cout << i << " ";  # 0 2 4 6 8 10
}

# Pętla z break:
for (int i = 0; i < 100; i++) {
    if (i == 10) break;
    cout << i << " ";  # 0 do 9
}

4. Pętla while

# while - sprawdza warunek NA POCZĄTKU:
int i = 0;
while (i < 5) {
    cout << i << " ";
    i++;
}

# do-while - sprawdza warunek NA KOŃCU:
int j = 0;
do {
    cout << j << " ";
    j++;
} while (j < 5);

# Różnica: do-while wykona się przynajmniej raz!
int x = 10;
while (x < 5) { cout << "Nie wypisze"; }  # Nigdy
do { cout << "Wypisze raz"; } while (x < 5);  # Raz

5. continue i break

# break - przerywa CAŁĄ pętlę:
for (int i = 0; i < 10; i++) {
    if (i == 5) break;
    cout << i << " ";  # 0 1 2 3 4
}

# continue - przerywa ITERACJĘ, przechodzi do następnej:
for (int i = 0; i < 5; i++) {
    if (i == 2) continue;
    cout << i << " ";  # 0 1 3 4 (pomija 2)
}

6. 🎮 Projekt: Gra w Zgadnij Liczbę

#include <iostream>
#include <cstdlib>
#include <ctime>
using namespace std;

int main() {
    srand(time(0));  # Inicjalizacja generatora losowego
    int wylosowana = rand() % 100 + 1;  # 1-100
    int zgadnij;
    int proby = 0;
    
    cout << "=== ZGADNIJ LICZBĘ (1-100) ===" << endl;
    
    do {
        cout << "Twoja liczba: ";
        cin >> zgadnij;
        proby++;
        
        if (zgadnij < wylosowana) {
            cout << "Za mało! Spróbuj większej." << endl;
        } else if (zgadnij > wylosowana) {
            cout << "Za dużo! Spróbuj mniejszej." << endl;
        }
    } while (zgadnij != wylosowana);
    
    cout << "\n🎉 BRAWO! Zgadłeś w " << proby << " próbach!" << endl;
    
    return 0;
}
