Zmienne i Typy Danych 📦

Podstawowe typy danych, deklaracje zmiennych i konwersje.

1. Deklaracja Zmiennych

W C++ musisz zadeklarować typ zmiennej przed użyciem:

int wiek = 25;           # liczba całkowita
double cena = 19.99;       # liczba zmiennoprzecinkowa
char litera = 'A';          # pojedynczy znak
bool czyAktywny = true;     # wartość logiczna
string imie = "Anna";       # tekst (wymaga <string>)

# Deklaracja bez wartości początkowej:
int liczba;
liczba = 10;  # przypisanie później

3. Konwersje Typów

# Konwersja niejawna (automatyczna):
int a = 10;
double b = a;  # int → double (10.0)

# Konwersja jawna (casting):
double x = 9.99;
int y = (int)x;  # C-style cast: 9
int z = static_cast<int>(x);  # C++ cast: 9

# Uważaj na utratę danych!
int duza = 1000000;
char mala = (char)duza;  # Przepełnienie!

4. Stałe (const)

# Stała - wartość nie może się zmienić:
const double PI = 3.14159;
const int MAX_GRACZY = 100;

# Próba zmiany spowoduje błąd kompilacji:
# PI = 3.14;  # BŁĄD!

# const jest lepsze niż #define:
const int PODATEK = 23;  # Ma typ, jest widoczne dla debuggera

5. Słowo kluczowe auto (C++11)

# Kompilator sam dedukuje typ:
auto x = 10;         # int
auto y = 3.14;       # double
auto tekst = "Hello"; # const char*

# Przydatne przy złożonych typach:
vector<map<string, int>> dana;
auto iterator = dana.begin();  # Zamiast pisać pełny typ

6. 🛠️ Projekt: Kalkulator BMI

#include <iostream>
#include <iomanip>  # do setprecision
using namespace std;

int main() {
    double wzrost_m, waga_kg;
    
    cout << "=== KALKULATOR BMI ===" << endl;
    cout << "Podaj wzrost (metry): ";
    cin >> wzrost_m;
    
    cout << "Podaj wagę (kg): ";
    cin >> waga_kg;
    
    double bmi = waga_kg / (wzrost_m * wzrost_m);
    
    cout << endl << "Twoje BMI: " << fixed << setprecision(2) << bmi << endl;
    
    if (bmi < 18.5) {
        cout << "Status: Niedowaga" << endl;
    } else if (bmi < 25) {
        cout << "Status: Waga prawidłowa" << endl;
    } else if (bmi < 30) {
        cout << "Status: Nadwaga" << endl;
    } else {
        cout << "Status: Otyłość" << endl;
    }
    
    return 0;
}
