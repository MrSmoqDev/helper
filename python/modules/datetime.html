Moduł Datetime 📅

Zaawansowana praca z datą i czasem – różnice, formatowanie, strefy czasowe.

1. Datetime vs Time

Moduł time (który już znasz) operuje na timestampach. datetime jest bardziej przyjazny: ma obiekty date, time i datetime, które łatwo porównywać i formatować.

from datetime import datetime, date, time, timedelta

# Aktualna data i czas
teraz = datetime.now()
print(teraz)  # 2024-01-15 14:30:45.123456

# Tylko data
dzis = date.today()
print(dzis)  # 2024-01-15

2. Tworzenie i Dostęp do Składowych

from datetime import datetime, date

# Ręczne tworzenie
urodziny = datetime(1990, 5, 15, 10, 30)  # YYYY, MM, DD, HH, MM
print(urodziny)

# Dostęp do części
print(urodziny.year)    # 1990
print(urodziny.month)   # 5
print(urodziny.day)     # 15
print(urodziny.hour)    # 10
print(urodziny.strftime("%A"))  # "Tuesday" (dzień tygodnia)

3. Formatowanie – strftime i strptime

strftime() zamienia datetime → string. strptime() string → datetime.

from datetime import datetime

teraz = datetime.now()

# Formatowanie na string
print(teraz.strftime("%Y-%m-%d"))      # 2024-01-15
print(teraz.strftime("%d.%m.%Y %H:%M")) # 15.01.2024 14:30
print(teraz.strftime("%A, %d %B %Y"))   # Monday, 15 January 2024

# Parsowanie stringa na datetime
tekst = "2024-12-25 18:00"
data = datetime.strptime(tekst, "%Y-%m-%d %H:%M")
print(data)  # 2024-12-25 18:00:00

4. Różnice Czasowe – timedelta

timedelta reprezentuje różnicę między dwiema datami. Idealne do obliczania wieku, deadline'ów, cooldownów.

from datetime import datetime, timedelta

teraz = datetime.now()

# Dodaj 7 dni
za_tydzien = teraz + timedelta(days=7)
print(f"Za tydzień: {za_tydzien.strftime('%Y-%m-%d')}")

# Odejmij 3 godziny
godziny_temu = teraz - timedelta(hours=3)

# Oblicz różnicę
urodziny = datetime(2024, 5, 15)
do_urodzin = urodziny - teraz
print(f"Do urodzin: {do_urodzin.days} dni")

# Przydatne jednostki timedelta
delta = timedelta(days=1, hours=2, minutes=30)
print(delta.total_seconds())  # Całkowita liczba sekund

5. 🎮 Projekt: System Cooldownów w Grze

from datetime import datetime, timedelta
import json

class Cooldown:
    """Prosty menedżer cooldownów"""
    def __init__(self, nazwa, sekundy):
        self.nazwa = nazwa
        self.czas_trwania = timedelta(seconds=sekundy)
        self.ostatnie_uzycie = None
    
    def mozna_uzy(self):
        if self.ostatnie_uzycie is None:
            return True
        return datetime.now() >= self.ostatnie_uzycie + self.czas_trwania
    
    def uzyj(self):
        if not self.mozna_uzy():
            pozostalo = (self.ostatnie_uzycie + self.czas_trwania) - datetime.now()
            return False, f"⏳ Poczekaj {pozostalo.seconds} sekund!"
        
        self.ostatnie_uzycie = datetime.now()
        return True, f"✅ {self.nazwa} użyte!"
    
    def do_json(self):
        """Serializacja do zapisu"""
        return {
            "nazwa": self.nazwa,
            "czas_trwania": self.czas_trwania.total_seconds(),
            "ostatnie_uzycie": self.ostatnie_uzycie.isoformat() if self.ostatnie_uzycie else None
        }
    
    @staticmethod
    def z_json(data):
        """Deserializacja z pliku"""
        cd = Cooldown(data["nazwa"], data["czas_trwania"])
        if data["ostatnie_uzycie"]:
            cd.ostatnie_uzycie = datetime.fromisoformat(data["ostatnie_uzycie"])
        return cd

# Demo
heal = Cooldown("Leczenie", sekundy=30)

for _ in range(3):
    ok, msg = heal.uzyj()
    print(msg)
    if ok:
        print("✨ Leczenie zastosowane!")
